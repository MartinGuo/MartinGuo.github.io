# 实现jQuery的基本思路


```javascript
 <script>
    (function(window){
        //----------核心结构start-----------
        var itcast = function (selector,context) {
            new itcast.prototype.init(selector,context);
        }
        itcast.prototype = {
            constructor: itcast,
            init: function (selector,context) {

            }
        }
        itcast.prototype.init.prototype=itcast.prototype;
        //------------核心结构end----------

        //-------------扩展-----------
        itcast.prototype.extend= function (obj) {
            var k;
            for(k in obj){
                this[k]=obj[k];
            };
        }
        //遍历
        itcast.extend({
            each: function (arr,callback) {
            }
        })
        //DOM操作模块
        itcast.extend({
            append: function () { },
            appendTo: function () { }
        })
        //样式操作模块
        itcast.extend({
            css: function () { },
            addClass: function () { },
            removeClass: function () { },
            hasClass: function () { }
        })
        //暴露itcast,使其在沙箱外调用
        window.itcast=itcast;
    })(window)
</script>
```
